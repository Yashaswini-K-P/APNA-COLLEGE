Qs2. Write a JavaScript function to extract unique characters from a string. 
    Example: str = “abcdabcdefgggh” ans = “abcdefgh”
Ans.
let str = "abcdabcdefgggh";
let ans ="";

function unique(str){
    for(char of str){
        if(ans.indexOf(char)==-1){
            ans+=char;
        }
    }
    return ans;
}
console.log(unique(str));

Qs3. Write a JavaScript function that accepts a list of country names as input and 
    returns the longest country name as output. 
    Example : country = ["Australia", "Germany", "United States of America"] output : 
    "United States of America"
Ans.
let country = ["Australia", "United States of America","Germany"];
function long(arr){
    let max=0;let i;let idx=0;
    for(i=0; i<arr.length; i++){
        if(arr[i].length>max){
            idx=i;
            max=arr[i].length;
        }
    }
    return arr[idx];
}
console.log(long(country));


Qs4. Write a JavaScript function to count the number of vowels in a String 
argument. 
Ans.
let str = "apnacollege"; 
function countVowels(str) { 
    let count = 0; 
 
    for (let i = 0; i < str.length; i++) { 
        if ( str.charAt(i) == "a" || 
        str.charAt(i) == "e" || 
        str.charAt(i) == "i" || 
        str.charAt(i) == "o" || 
        str.charAt(i) == "u" ) { 
            count++; 
        } 
    } 
    return count; 
}


Qs5. Write a JavaScript function to generate a random number within a range 
    (start, end).
Ans.
function random(start, end){
    let num =Math.floor(Math.random()*(end-start)+start);
    return num;
}

console.log(random(5,15));
